#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    pid_t pid;

    pid = fork();  // Create new process

    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    }
    else if (pid == 0) {
        // Child process
        printf("Child Process\n");
        printf("Child PID: %d\n", getpid());

        execlp("/bin/ls", "ls", NULL); // Execute ls command
        printf("Exec failed\n");
        exit(0);
    }
    else {
        // Parent process
        wait(NULL);  // Wait for child to finish
        printf("Parent Process\n");
        printf("Parent PID: %d\n", getpid());
    }

    return 0;
}

/////////////////////////////////////////////////////////////////1. First Come First Serve

#include <stdio.h>

int main() {
    int n, bt[20], wt[20], tat[20];
    int i;
    float awt = 0, atat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    printf("Enter burst time for each process:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &bt[i]);
    }

    wt[0] = 0;
    for(i = 1; i < n; i++) {
        wt[i] = wt[i - 1] + bt[i - 1];
    }

    for(i = 0; i < n; i++) {
        tat[i] = wt[i] + bt[i];
        awt += wt[i];
        atat += tat[i];
    }

    printf("\nProcess\tBT\tWT\tTAT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\n", i + 1, bt[i], wt[i], tat[i]);
    }

    printf("\nAverage Waiting Time: %.2f", awt / n);
    printf("\nAverage Turnaround Time: %.2f\n", atat / n);

    return 0;
}

/////////////////////////////////////////////////////////////////2. Shortest Job First

#include <stdio.h>

int main() {
    int n, bt[20], wt[20], tat[20], i, j, temp;
    float awt = 0, atat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    printf("Enter burst time for each process:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &bt[i]);
    }

    // Sorting burst times (SJF)
    for(i = 0; i < n - 1; i++) {
        for(j = i + 1; j < n; j++) {
            if(bt[i] > bt[j]) {
                temp = bt[i];
                bt[i] = bt[j];
                bt[j] = temp;
            }
        }
    }

    wt[0] = 0;
    for(i = 1; i < n; i++) {
        wt[i] = wt[i - 1] + bt[i - 1];
    }

    for(i = 0; i < n; i++) {
        tat[i] = wt[i] + bt[i];
        awt += wt[i];
        atat += tat[i];
    }

    printf("\nProcess\tBT\tWT\tTAT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\n", i + 1, bt[i], wt[i], tat[i]);
    }

    printf("\nAverage Waiting Time: %.2f", awt / n);
    printf("\nAverage Turnaround Time: %.2f\n", atat / n);

    return 0;
}

/////////////////////////////////////////////////////////////////3. Priority Scheduling

#include <stdio.h>

int main() {
    int n, bt[20], p[20], pr[20], wt[20], tat[20], i, j, temp;
    float awt = 0, atat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    printf("Enter burst time and priority for each process:\n");
    for(i = 0; i < n; i++) {
        scanf("%d%d", &bt[i], &pr[i]);
        p[i] = i + 1;
    }

    // Sorting based on priority
    for(i = 0; i < n - 1; i++) {
        for(j = i + 1; j < n; j++) {
            if(pr[i] > pr[j]) {
                temp = pr[i];
                pr[i] = pr[j];
                pr[j] = temp;

                temp = bt[i];
                bt[i] = bt[j];
                bt[j] = temp;

                temp = p[i];
                p[i] = p[j];
                p[j] = temp;
            }
        }
    }

    wt[0] = 0;
    for(i = 1; i < n; i++) {
        wt[i] = wt[i - 1] + bt[i - 1];
    }

    for(i = 0; i < n; i++) {
        tat[i] = wt[i] + bt[i];
        awt += wt[i];
        atat += tat[i];
    }

    printf("\nProcess\tBT\tPriority\tWT\tTAT\n");
    for(i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\n", p[i], bt[i], pr[i], wt[i], tat[i]);
    }

    printf("\nAverage Waiting Time: %.2f", awt / n);
    printf("\nAverage Turnaround Time: %.2f\n", atat / n);

    return 0;
}

#include <stdio.h>

int main() {
    int n, i;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int bt[n], queue[n];

    for(i = 0; i < n; i++) {
        printf("Enter Burst Time and Queue (1-High, 2-Low) for P%d: ", i + 1);
        scanf("%d %d", &bt[i], &queue[i]);
    }

    printf("\n--- Queue 1 (High Priority) ---\n");
    for(i = 0; i < n; i++)
        if(queue[i] == 1)
            printf("P%d executed (BT=%d)\n", i + 1, bt[i]);

    printf("\n--- Queue 2 (Low Priority) ---\n");
    for(i = 0; i < n; i++)
        if(queue[i] == 2)
            printf("P%d executed (BT=%d)\n", i + 1, bt[i]);

    return 0;
}

/////////////////////////////////////////////////////////////////1. Internal Fragmentation

#include <stdio.h>

int main() {
    int blockSize[10], processSize[10], n, i, totalInternalFragmentation = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    printf("Enter block sizes:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &blockSize[i]);
    }

    printf("Enter process sizes:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &processSize[i]);
    }

    for(i = 0; i < n; i++) {
        if(processSize[i] <= blockSize[i]) {
            totalInternalFragmentation += blockSize[i] - processSize[i];
        }
    }

    printf("\nTotal Internal Fragmentation: %d\n", totalInternalFragmentation);

    return 0;
}
