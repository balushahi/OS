//////////////////////////////////////////////////////1. Process Management Program

#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    pid_t pid;

    pid = fork();  // Create new process

    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    }
    else if (pid == 0) {
        // Child process
        printf("Child Process\n");
        printf("Child PID: %d\n", getpid());

        execlp("/bin/ls", "ls", NULL); // Execute ls command
        printf("Exec failed\n");
        exit(0);
    }
    else {
        // Parent process
        wait(NULL);  // Wait for child to finish
        printf("Parent Process\n");
        printf("Parent PID: %d\n", getpid());
    }

    return 0;
}

////////////////////////////////////////////////////////2. File Management Program

#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
    int fd;
    char buffer[50];

    fd = open("test.txt", O_CREAT | O_RDWR, 0644);

    write(fd, "Hello Operating System", 23);

    lseek(fd, 0, SEEK_SET); // Move pointer to beginning

    read(fd, buffer, 23);
    buffer[23] = '\0';

    printf("File Content: %s\n", buffer);

    close(fd);

    return 0;
}

////////////////////////////////////////////////////////3. Input/Output Management Program

#include <unistd.h>

int main() {
    char buffer[100];
    int n;

    write(1, "Enter some text: ", 17); // 1 = stdout

    n = read(0, buffer, 100); // 0 = stdin

    write(1, "You entered: ", 13);

    return 0;
}
